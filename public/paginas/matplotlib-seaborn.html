Visualizacoes com Matplotlib e Seaborn

Crie graficos profissionais em Python para explorar dados e comunicar insights.

Matplotlib vs Seaborn

Matplotlib

Biblioteca base de visualizacao. Maximo controle e personalizacao. Mais verbosa.

  • Controle total sobre cada elemento
  • Base para outras bibliotecas
  • Mais codigo necessario

Seaborn

Construida sobre Matplotlib. Graficos bonitos com menos codigo. Foco em estatistica.

  • Graficos bonitos por padrao
  • Ideal para exploracao rapida
  • Integrado com Pandas
import matplotlib.pyplot as plt import seaborn as sns import pandas as pd # Configuracao recomendada plt.style.use('seaborn-v0_8-whitegrid') sns.set_palette('husl') plt.rcParams['figure.figsize'] = (10, 6) plt.rcParams['font.size'] = 12

Graficos com Matplotlib

Grafico de Linha

fig, ax = plt.subplots(figsize=(10, 6)) ax.plot(df['data'], df['vendas'], label='Vendas', color='blue', linewidth=2) ax.plot(df['data'], df['meta'], label='Meta', color='red', linestyle='--') ax.set_title('Vendas vs Meta ao Longo do Tempo', fontsize=14) ax.set_xlabel('Data') ax.set_ylabel('Valor (R$)') ax.legend() ax.grid(True, alpha=0.3) plt.tight_layout() plt.savefig('grafico.png', dpi=300) plt.show()

Grafico de Barras

fig, ax = plt.subplots(figsize=(10, 6)) cores = ['#3498db', '#e74c3c', '#2ecc71', '#f39c12', '#9b59b6'] ax.bar(df['regiao'], df['vendas'], color=cores) # Adicionar valores nas barras for i, v in enumerate(df['vendas']): ax.text(i, v + 1000, f'R$ {v:,.0f}', ha='center', fontsize=10) ax.set_title('Vendas por Regiao') ax.set_ylabel('Vendas (R$)') plt.xticks(rotation=45) plt.tight_layout() plt.show()

Subplots (Multiplos Graficos)

fig, axes = plt.subplots(2, 2, figsize=(12, 10)) # Grafico 1 axes[0, 0].plot(df['data'], df['vendas']) axes[0, 0].set_title('Tendencia de Vendas') # Grafico 2 axes[0, 1].bar(df['produto'], df['quantidade']) axes[0, 1].set_title('Quantidade por Produto') # Grafico 3 axes[1, 0].scatter(df['preco'], df['vendas']) axes[1, 0].set_title('Preco vs Vendas') # Grafico 4 axes[1, 1].pie(df['vendas'], labels=df['regiao'], autopct='%1.1f%%') axes[1, 1].set_title('Distribuicao por Regiao') plt.tight_layout() plt.show()

Graficos com Seaborn

Graficos Categoricos

# Barras com media e intervalo de confianca sns.barplot(data=df, x='regiao', y='vendas', hue='categoria') # Box plot sns.boxplot(data=df, x='regiao', y='vendas') # Violin plot (box + distribuicao) sns.violinplot(data=df, x='categoria', y='valor') # Count plot (contagem) sns.countplot(data=df, x='categoria', hue='status')

Graficos de Distribuicao

# Histograma com KDE sns.histplot(data=df, x='idade', kde=True, bins=30) # KDE plot (densidade) sns.kdeplot(data=df, x='salario', hue='departamento', fill=True) # ECDF (distribuicao acumulada) sns.ecdfplot(data=df, x='valor')

Graficos de Relacao

# Scatter com regressao sns.regplot(data=df, x='idade', y='salario') # Scatter com cores por categoria sns.scatterplot(data=df, x='preco', y='vendas', hue='categoria', size='quantidade') # Pair plot (multiplas variaveis) sns.pairplot(df[['idade', 'salario', 'experiencia']], diag_kind='kde') # Correlation heatmap correlacao = df[['preco', 'quantidade', 'desconto', 'lucro']].corr() sns.heatmap(correlacao, annot=True, cmap='coolwarm', center=0)

Personalizacao Avancada

# Tema e estilo sns.set_theme(style='whitegrid', palette='deep') # Cores personalizadas cores_empresa = ['#003366', '#0066CC', '#3399FF', '#66CCFF'] sns.set_palette(cores_empresa) # Figura com estilo profissional fig, ax = plt.subplots(figsize=(12, 6)) sns.barplot(data=df, x='mes', y='vendas', color='#003366', ax=ax) ax.set_title('Vendas Mensais 2024', fontsize=16, fontweight='bold', pad=20) ax.set_xlabel('Mes', fontsize=12) ax.set_ylabel('Vendas (R$ milhoes)', fontsize=12) # Remover bordas ax.spines['top'].set_visible(False) ax.spines['right'].set_visible(False) # Formatacao do eixo Y ax.yaxis.set_major_formatter(plt.FuncFormatter(lambda x, p: f'{x/1e6:.1f}M')) # Adicionar fonte plt.text(0.99, 0.01, 'Fonte: Dados internos', transform=ax.transAxes, ha='right', fontsize=8, color='gray') plt.tight_layout() plt.show()
